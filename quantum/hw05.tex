\section*{Problem 2.34}

\begin{enumproblem}

% part A
\item \textit{
Write down an expression for the classical action for a simple harmonic oscillator for a finite time interval.
}


% part B
\item \textit{
Construct $\braket{x_n, t_n}{x_{n-1},t_{n-1}}$ for a simple harmonic oscillator using Feynman's prescription for $t_n - t_{n-1} = \Delta t$ small. Keeping only terms up to order $(\Delta t)^2$, show that it is in complete agreement with the $t-t_0 \rightarrow 0$ limit of the propagator given by (2.6.26),
\begin{align}
K(\v x'', t, \v x', t_0) &= \sum_{a'} \braket{\v x''}{a'} \braket{a'}{\v x'} \exp \left[ \frac{-i E_{a'} (t-t_0)}{\hbar} \right] \\
	&= \sum_{a'} \bra{\v x''}{\exp \left( \frac{-i H t}{\hbar} \right)}\ket{a'} \bra{a'}{\exp \left( \frac{i H t_0}{\hbar} \right)} \ket{\v x'} \\
	&= \braket{\v x'', t}{\v x', t_0}.
\end{align}
}


\end{enumproblem}



\section*{Problem 2.37}

m \od[2]{\v x}{t} = \od{\Pi}{t} = e \left[ \v E + \frac{1}{2c} \left( \od{\v x}{t} \times \v B - \v B \times \od{\v x}{t} \right) \right].

\v j = \left( \frac{\hbar}{m} \right) \Im(\psi^* \v \nabla' \psi) - \left( \frac{e}{mc} \right) \v A \abs{\psi}^2.

\section*{Problem 2.38}
\textit{
Consider the Hamiltonian of a spinless particle of charge $e$. In the presence of a static magnetic field, the interaction terms can be generated by 
\begin{eqn}
\v p_\text{operator} \rightarrow \v p_\text{operator} - \frac{e \v A}{c},
\end{eqn}
where $\v A$ is the appropriate vector potential. Suppose, for simplicity, that the magnetic field $\v B$ is uniform in the positive $z$-direction. Prove that the above prescription indeed leads to the correct expression for the interaction of the orbital magnetic moment $(e/2mc) \v L$ with the magnetic field $\v B$. Show that there is also an extra term proportional to $B^2(x^2+y^2)$, and comment briefly on its physical significance. 
}



\section*{Problem 2.39}
\textit{
An electron moves in the presence of a uniform magnetic field in the $z$-direction ($\v B = B \uv z$).
}

\begin{enumproblem}

% part A
\item \textit{
Evaluate $[\Pi_x, \Pi_y]$, where
\begin{eqn}
\Pi_x \equiv p_x - \frac{e A_x}{c}, \qquad
\Pi_y \equiv p_y - \frac{e A_y}{c}.
\end{eqn}
}


% part B
\item \textit{
By comparing the Hamiltonian and the commutation relation obtain in (a) with those of the one-dimensional oscillator problem, show how we can immediately write the energy eigenvalues as
\begin{eqn}
E_{k,n} = \frac{\hbar^2 k^2}{2m} + \left( \frac{\abs{eB} \hbar}{mc} \right) \left( n + \frac{1}{2} \right),
\end{eqn}
where $\hbar k$ is the continuous eigenvalue of the $p_z$ operator and $n$ is a nonnegative integer including zero. 
}


\end{enumproblem}
