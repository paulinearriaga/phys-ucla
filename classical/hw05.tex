\section*{Problem 1 (15 pts)}
\textit{
This problem concerns the Runge-Lanz vector and its algebra under Poisson brackets. The Runge-Lenz vector is defined as
\begin{eqn}
\v{K} = \frac{1}{m} \v p \times \v L - \frac{\v r}{r}.
\end{eqn}
}

\begin{enumproblem}

% part A
\item \textit{
Show that the following Poisson bracket is given by
\begin{eqn}
\cbr{L_i, K_j} = \epsilon_{ijk} K_k.
\end{eqn}
}


% part B
\item \textit{
Show that
\begin{eqn}
\cbr{K_i, K_j} = - \frac{2}{m} \left( \frac{p^2}{2m} - \frac{1}{r} \right) \epsilon_{ijk} L_k.
\end{eqn}
}


% part C
\item \textit{
Show that for a Hamiltonian of the form
\begin{eqn}
H = \frac{p^2}{2m} + V(r)
\end{eqn}
one has 
\begin{eqn}
[\v K, H] = \frac{\v r \times (\v p \times \v r)}{m} \left( \frac{-1 + r^2 V'(r)}{r^3} \right).
\end{eqn}
}


\end{enumproblem}

}

\end{enumproblem}

\section*{Problem 4 (10 pts)}
\textit{
Consider a dynamical flow which is defined on $\R^n$ with $\v x = (x_1, \dots x_n)$ and a vector field $\v v(\v x)$ as follows
\begin{eqn}
\od{}{t} \v x(t) = \v v(\v x(t)).
\end{eqn}
}

\begin{enumproblem}

% part A
\item \textit{
Solving the differential equations and finding $\v x(t)$ defines a map from $\R^n \rightarrow \R^n$. Show that this map is volume preserving if the vector field $\v v$ is divergence free, i.e.
\begin{eqn}
\sum_i \pd{v_i}{x_i} = 0.
\end{eqn}
}


% part B
\item \textit{
Show that by setting $\v x = \set{q_1, \dots, q_n, p_1, \dots, p_n}$ and 
\begin{eqn}
v^i = \sum_j J^i_j \pd{H}{{x^j}}, \qquad
J = \pmat{0 & 1_n \\ -1_n & 0},
\end{eqn}
the vector field $\v v$ is divergence free if Hamilton's equations hold. Hence you have proved Liouville's theorem.
}


\end{enumproblem}
